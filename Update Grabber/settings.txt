[AUTOMATIC_INCLUDED:True] #(Case sensitive boolean) While updates can always be called using the "update()" method, changing this boolean to True will result in UG updating files as soon as the attached program is ran. This will require a forced automatic restart by UG when the program starts.
[CHECK_FOR_CHANGES:True] #(Case sensitive boolean) If True, UG will only update files IF it detects changes between versions.
[NOTIFY_USER_ON_RESTART:True] #(Case sensitive boolean) If True, the user will be notified when UG restarts the program to update. This applies to automatic and manual updates.
--------------------------

[GITHUB_TARGET_REPOSITORY:"CHANGE_ME"] #The name of the repository you want to grab. Example: "UpdateGrabber"

[GITHUB_USERNAME:"CHANGE_ME"] #The GitHub username attached to GITHUB_TARGET_REPOSITORY, this is used to find the exact URL of the repository. Example: "Turtle"

[GITHUB_TARGET_BRANCH:"CHANGE_ME"] #The selected branch that will be copied. Usually named "main" or "master".

[GLOBAL_DIRECTORY_PATH:r"CHANGE_ME"] #This is the global path to the directory being updated. Example: r"C:\Users\Turtle\OneDrive\Documents"
